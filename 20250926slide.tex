\begin{frame}\frametitle{グラフって、何？}

\begin{definition}[有向グラフ]
有限集合$V$および$V$上の二項関係の部分集合$E$の組$(V, E)$を\structure{有向グラフ}と呼ぶ。
このとき、$V$を\structure{頂点集合}、$E$を\structure{辺集合}と呼ぶ。

また、$V$の要素をグラフ$G$の\structure{頂点}、$E$の要素を$G$の\structure{辺}と呼ぶ。
\end{definition}

\end{frame}
